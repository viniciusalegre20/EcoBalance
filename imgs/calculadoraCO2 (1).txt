<script> function calcularEmissoes() { const valorEletricidade = parseFloat(document.getElementById('valor-eletricidade').value); const valorGas = parseFloat(document.getElementById('valor-gas').value); const kmTransporte = parseFloat(document.getElementById('km-transporte').value); const combustivel = document.getElementById('selection-combustivel').value; const transporte = document.getElementById('selection-transport').value; // Fatores de emissão de CO2 const fatoresEmissao = { gasolina: 2.3, // kg CO2 por litro etanol: 0.82, diesel: 2.6 }; let resultado = "Resultados de Emissões de CO₂: <br><br>"; let emissaoTotal = 0; // Transporte if (transporte === "1" && combustivel !== "0" && kmTransporte > 0) { let fatorCombustivel; if (combustivel === "1") fatorCombustivel = fatoresEmissao.gasolina; else if (combustivel === "2") fatorCombustivel = fatoresEmissao.etanol; else if (combustivel === "3") fatorCombustivel = fatoresEmissao.diesel; const emissaoTransporte = kmTransporte * fatorCombustivel; emissaoTotal += emissaoTransporte; resultado += `Transporte - Emissão de CO₂: ${emissaoTransporte.toFixed(2)} kg <br><br>`; } // Eletricidade (assumindo kWh) if (valorEletricidade > 0) { const emissaoEletricidade = valorEletricidade * 0.00051; emissaoTotal += emissaoEletricidade; resultado += `Eletricidade - Emissão de CO₂: ${emissaoEletricidade.toFixed(2)} kg <br><br>`; } // Gás if (valorGas > 0) { const emissaoGas = valorGas * 2.3; emissaoTotal += emissaoGas; resultado += `Gás de Cozinha - Emissão de CO₂: ${emissaoGas.toFixed(2)} kg <br><br>`; } // Cálculo da compensação financeira sugerida const valorDoacao = (emissaoTotal / 1000) * 50; resultado += `<strong>Total estimado de CO₂: ${emissaoTotal.toFixed(2)} kg</strong><br>`; resultado += `<strong>Valor sugerido para compensação: R$ ${valorDoacao.toFixed(2)}</strong>`; document.getElementById('resultado').innerHTML = resultado; } </script>
