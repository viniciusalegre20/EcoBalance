<script>
        function calcularEmissoes() {
            const valorEletricidade = parseFloat(document.getElementById('valor-eletricidade').value);
            const valorGas = parseFloat(document.getElementById('valor-gas').value);
            const kmTransporte = parseFloat(document.getElementById('km-transporte').value);
            const combustivel = document.getElementById('selection-combustivel').value;
            const transporte = document.getElementById('selection-transport').value;

            // Fatores de emissão de CO2
            const fatoresEmissao = {
                gasolina: 2.3, // kg CO2 por litro
                etanol: 0.82,
                diesel: 2.6
            };

            let resultado = "Resultados de Emissões de CO₂: <br><br>";

            // Cálculo para transporte
            if (transporte === "1" && combustivel !== "0" && kmTransporte > 0) {
                let fatorCombustivel;
                if (combustivel === "1") fatorCombustivel = fatoresEmissao.gasolina;
                else if (combustivel === "2") fatorCombustivel = fatoresEmissao.etanol;
                else if (combustivel === "3") fatorCombustivel = fatoresEmissao.diesel;

                const emissaoTransporte = kmTransporte * fatorCombustivel;
                resultado += `Transporte (Carro/Moto) - Emissão de CO₂: ${emissaoTransporte.toFixed(2)} kg <br><br>`;
            }

            // Cálculo de eletricidade (assumindo 0.00051 kg CO₂ por kWh)
            if (valorEletricidade > 0) {
                const emissaoEletricidade = valorEletricidade * 0.00051; // fator de emissão para kWh
                resultado += `Eletricidade - Emissão de CO₂: ${emissaoEletricidade.toFixed(2)} kg <br><br>`;
            }

            // Cálculo de gás (assumindo 2.3 kg CO₂ por m³)
            if (valorGas > 0) {
                const emissaoGas = valorGas * 2.3;
                resultado += `Gás de Cozinha - Emissão de CO₂: ${emissaoGas.toFixed(2)} kg <br><br>`;
            }

            document.getElementById('resultado').innerHTML = resultado;
        }
    </script>
